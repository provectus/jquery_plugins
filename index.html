Selects

Description:
Plugin replaces default selects with more beautiful analogues. It monitors changes in original element and applies them on itself.
Usage:
<select class="s-select class-to-find" placeholder="placeholder">
  <optgroup label="Number Values">
   <option>Value 1</option>
   <option>Value 2</option>
  </optgroup>
  <option>Value A</option>
  <option>Value B</option>
</select>
$(function () {
  //initialize
  $('select.class-to-find').selects(options);
  //append item
  $('select.class-to-find').append('<option>Value</option>');
  //remove element
  $('select.class-to-find').find('.select-of-option').remove()
  //getting value
  $('select.class-to-find').val();
  //setting value
  $('select.class-to-find').val('<value of selecting option>');
  //disable
  $('select.class-to-find').prop('disabled', true);
  //enable
  $('select.class-to-find').prop('disabled', false);
});
Options:
{
  input: true/false // text field in select, default: false
}

Datepicker

Description:
Plugin draws pretty calendar for choosing date.
Usage:
<input type="date" class="s-calendar class-to-find" />
$(function () {
  $('input.class-to-find').datepicker(options);
});
Options:
{
  week_starts: [0..6] // day of week, when week starts (0 - Sunday, 6 - Saturday), default: 0
  access_past: true/false // accept or deny choose dates in past, default: true
  _default: "YYYY-MM-DD" // predefined date, default: today
}
